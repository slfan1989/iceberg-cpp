168
169 private:

171 static Result<PartitionSpecsMap> InitPartitionSpecsMap(const TableMetadata* metadata);
172 static Result<SortOrdersMap> InitSortOrdersMap(const TableMetadata* metadata);
173 static Result<SnapshotsMap> InitSnapshotMap(const TableMetadata* metadata);
174
175 const TableMetadata* metadata_;
