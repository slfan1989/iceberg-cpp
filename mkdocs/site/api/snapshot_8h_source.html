46ICEBERG_EXPORT constexpr std::string_view ToString(SnapshotRefType type) noexcept {
47 switch (type) {
48 case SnapshotRefType::kBranch:
49 return "branch";
50 case SnapshotRefType::kTag:
51 return "tag";
52 }
53 std::unreachable();
54}
56ICEBERG_EXPORT constexpr Result<SnapshotRefType> SnapshotRefTypeFromString(
57 std::string_view str) noexcept {
58 if (str == "branch") return SnapshotRefType::kBranch;
59 if (str == "tag") return SnapshotRefType::kTag;
60 return InvalidArgument("Invalid snapshot reference type: {}", str);
61}
