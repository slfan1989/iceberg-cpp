146 public:
147 explicit TableMetadataCache(const TableMetadata* metadata) : metadata_(metadata) {}
148
149 template <typename T>
150 using ByIdMap = std::unordered_map<int32_t, std::shared_ptr<T>>;
151 using SchemasMap = ByIdMap<Schema>;
152 using PartitionSpecsMap = ByIdMap<PartitionSpec>;
153 using SortOrdersMap = ByIdMap<SortOrder>;
154 using SnapshotsMap = std::unordered_map<int64_t, std::shared_ptr<Snapshot>>;
155 using SchemasMapRef = std::reference_wrapper<const SchemasMap>;
156 using PartitionSpecsMapRef = std::reference_wrapper<const PartitionSpecsMap>;
157 using SortOrdersMapRef = std::reference_wrapper<const SortOrdersMap>;
158 using SnapshotsMapRef = std::reference_wrapper<const SnapshotsMap>;
159
160 Result<SchemasMapRef> GetSchemasById() const;
161 Result<PartitionSpecsMapRef> GetPartitionSpecsById() const;
162 Result<SortOrdersMapRef> GetSortOrdersById() const;
163 Result<SnapshotsMapRef> GetSnapshotsById() const;
164
165 private:
166 static Result<SchemasMap> InitSchemasMap(const TableMetadata* metadata);
167 static Result<PartitionSpecsMap> InitPartitionSpecsMap(const TableMetadata* metadata);
168 static Result<SortOrdersMap> InitSortOrdersMap(const TableMetadata* metadata);
169 static Result<SnapshotsMap> InitSnapshotMap(const TableMetadata* metadata);
170
171 const TableMetadata* metadata_;
172 Lazy<InitSchemasMap> schemas_map_;
173 Lazy<InitPartitionSpecsMap> partition_specs_map_;
174 Lazy<InitSortOrdersMap> sort_orders_map_;
175 Lazy<InitSnapshotMap> snapshot_map_;
176};
